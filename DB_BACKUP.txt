-- //--------------- 表作成 ----------------------------------

-- KEY は自動生成じゃなくても良かった・・
CREATE TABLE  "ABILITY_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"ABILITY_NAME" VARCHAR(100)
   );

CREATE TABLE  "ALIGNMENT_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"ALIGNMENT_NAME" VARCHAR(100),
	"ALIGNMENT_SHORT_NAME" VARCHAR(100)
   );

CREATE TABLE  "BONUS_RANK_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"RANK_NAME" VARCHAR(100)
   );

CREATE TABLE "CAMPAIGN_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"CAMPAIGN_NAME" VARCHAR(100),
	"MASTER" VARCHAR(100)
   );

CREATE TABLE  "GENDER_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"GENDER" VARCHAR(100)
   );

CREATE TABLE "DICE_MASTER"
  (     "ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        "NAME" varchar(20),
        "TYPE" INT
   );

CREATE TABLE  "RELIGION_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"RELIGION_NAME" VARCHAR(100),
	"ALIGNMENT_ID" INT,
	"DESCRIPTION" VARCHAR(4000),
        FOREIGN KEY (ALIGNMENT_ID) REFERENCES ALIGNMENT_MASTER(ID)
   );

CREATE TABLE  "SIZE_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"SIZE_NAME" VARCHAR(100),
	"AC_MODIFIER" INT
   );

CREATE TABLE "SKILL_MASTER"
  ( "ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    "SKILL_NAME" VARCHAR(100),
    "ACCEPT_NO_RANK" INT,
    "ABILITY_ID" INT,
    "ARMOR_CHECK" INT,

    FOREIGN KEY (ABILITY_ID) REFERENCES ABILITY_MASTER(ID)
  );

CREATE TABLE "SAVE_MASTER"
  ( "ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    "SAVE_NAME" VARCHAR(100),
    "ABILITY_ID" INT,
     FOREIGN KEY (ABILITY_ID) REFERENCES ABILITY_MASTER(ID)
  );

CREATE TABLE "RACE_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"RACE_NAME" VARCHAR(100),
	"SIZE_ID" INT,
        "SPEED" INT DEFAULT 0,

	"DESCRIPTION" VARCHAR(4000),
        FOREIGN KEY (SIZE_ID) REFERENCES SIZE_MASTER(ID)
    );

CREATE TABLE  "CLASS_MASTER"
   (	"ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	"CLASS_NAME" VARCHAR(100),
	"BASE_ATTACK_RANK_ID" INT,
	"DESCRIPTION" VARCHAR(4000),
        "HIT_DICE_TYPE" INT,
        "SKILL_POINT" INTEGER,

        FOREIGN KEY (BASE_ATTACK_RANK_ID) REFERENCES BONUS_RANK_MASTER(ID),
        FOREIGN KEY (HIT_DICE_TYPE) REFERENCES DICE_MASTER(ID)
   );

CREATE TABLE "CLASS_ABILITY_MASTER"
   (    "CLASS_ID" INT,
        "ABILITY_ID" INT,
        "RANK_ID" INT,
        PRIMARY KEY (CLASS_ID, ABILITY_ID),
        FOREIGN KEY (CLASS_ID) REFERENCES CLASS_MASTER(ID) ON DELETE CASCADE,
        FOREIGN KEY (ABILITY_ID) REFERENCES ABILITY_MASTER(ID),
        FOREIGN KEY (RANK_ID) REFERENCES BONUS_RANK_MASTER(ID)
   );

CREATE TABLE "CLASS_SAVE_MASTER"
   (    "CLASS_ID" INT,
        "SAVE_ID" INT,
        "RANK_ID" INT,
        PRIMARY KEY (CLASS_ID, SAVE_ID),
        FOREIGN KEY (CLASS_ID) REFERENCES CLASS_MASTER(ID) ON DELETE CASCADE,
        FOREIGN KEY (SAVE_ID) REFERENCES SAVE_MASTER(ID),
        FOREIGN KEY (RANK_ID) REFERENCES BONUS_RANK_MASTER(ID)
   );

CREATE TABLE "CLASS_SKILL_MASTER"
  ("CLASS_ID" INT,
   "SKILL_ID" INT,
     PRIMARY KEY (CLASS_ID, SKILL_ID),
     FOREIGN KEY (CLASS_ID) REFERENCES CLASS_MASTER(ID) ON DELETE CASCADE
   );

CREATE TABLE "RACE_ABILITY_MASTER"
   (    "RACE_ID"  INT,
        "ABILITY_ID" INT,
        "MODIFIER" INT DEFAULT 0,
        PRIMARY KEY (RACE_ID, ABILITY_ID),
        FOREIGN KEY (RACE_ID) REFERENCES RACE_MASTER(ID) ON DELETE CASCADE,
        FOREIGN KEY (ABILITY_ID) REFERENCES ABILITY_MASTER(ID)
   );

CREATE TABLE "RACE_SAVE_MASTER"
  (    "RACE_ID" INT,
       "SAVE_ID" INT,
        "MODIFIER" INT DEFAULT 0,
        PRIMARY KEY (RACE_ID, SAVE_ID),
        FOREIGN KEY (RACE_ID) REFERENCES RACE_MASTER(ID) ON DELETE CASCADE,
        FOREIGN KEY (SAVE_ID) REFERENCES SAVE_MASTER(ID)
   );


CREATE TABLE "CHARACTER_RECORD"
   (    "ID" INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        "SAVE_TIME" TIMESTAMP,
        "CAMPAIGN_ID" INT,
     	"SHEET_ID" INT,
    	"CHARACTER_NAME" VARCHAR(100),
	    "PLAYER_NAME" VARCHAR(100),
    	"EXPERIENCE" INT DEFAULT 0,
	    "ALIGNMENT_ID" INT,
	    "GENDER_ID" INT,
    	"HEIGHT" INT DEFAULT 0,
    	"WEIGHT" INT DEFAULT 0,
    	"DESCRIPTION" VARCHAR(4000),
    	"RELIGION_ID" INT,
        "RACE_ID" INT,
        "INITIATIVE_MISC_MODIFIER" INT DEFAULT 0,
        "INITIATIVE_FEAT_MODIFIER" INT DEFAULT 0,
        "SPEED_MISC_MODIFIER" INT DEFAULT 0,
        "SPEED_FEAT_MODIFIER" INT DEFAULT 0,
        "HAIR_COLOR" VARCHAR(200),
        "SKIN_COLOR" VARCHAR(200),
        "EYE_COLOR"  VARCHAR(200),
        "DAMAGE_REDUCTION" INT,
        "SPELL_RESISTANCE" INT,
        "LANGUAGE"  VARCHAR(400),
        "FEAT_DESCRIPTION" VARCHAR(4000),
        "ATTACK_DESCRIPTION" VARCHAR(4000),
        "SPELL_DESCRIPTION" VARCHAR(4000),
        "ITEM_DESCRIPTION" VARCHAR(4000),
        "DEFENCE_DESCRIPTION" VARCHAR(4000),
        "UPDATE_DESCRIPTION" VARCHAR(4000),
        "AC_ARMOR" INT,
        "AC_SHIELD" INT,
        "AC_MISC_MOD" INT,
        "AGE" INT,

         FOREIGN KEY (ALIGNMENT_ID) REFERENCES ALIGNMENT_MASTER(ID),
         FOREIGN KEY (RELIGION_ID) REFERENCES RELIGION_MASTER(ID),
         FOREIGN KEY (CAMPAIGN_ID) REFERENCES CAMPAIGN_MASTER(ID),
         FOREIGN KEY (RACE_ID) REFERENCES RACE_MASTER(ID),
         FOREIGN KEY (GENDER_ID) REFERENCES GENDER_MASTER(ID)
   );

CREATE TABLE "CHARACTER_GROWTH_RECORD"
   (	"CHARACTER_ID" INT,
    	"CHARACTER_LEVEL" INT,
	    "CLASS_ID" INT,
	    "ABILITY_ENHANCEMENT" INT DEFAULT 0,
        "HIT_POINT" INT DEFAULT 0,
        "UPDATE_DESCRIPTION" VARCHAR(4000),

        PRIMARY KEY (CHARACTER_ID, CHARACTER_LEVEL),
        FOREIGN KEY (CHARACTER_ID) REFERENCES CHARACTER_RECORD(ID) ON DELETE CASCADE,
        FOREIGN KEY (CLASS_ID) REFERENCES CLASS_MASTER(ID)
   );

CREATE TABLE "CHARACTER_SKILL_GROWTH_RECORD"
   (	"CHARACTER_ID" INT,
	"CHARACTER_LEVEL" INT,
        "SKILL_ID" INT,
        "SKILL_POINT" INT DEFAULT 0,
	"DESCRIPTION" VARCHAR(400),

        PRIMARY KEY (CHARACTER_ID, CHARACTER_LEVEL, SKILL_ID),
        FOREIGN KEY (CHARACTER_ID) REFERENCES CHARACTER_RECORD(ID) ON DELETE CASCADE,
        FOREIGN KEY (SKILL_ID) REFERENCES SKILL_MASTER(ID)
   );

CREATE TABLE "CHARACTER_SKILL_RECORD"
   (	"CHARACTER_ID" INT,
        "SKILL_ID" INT,
        "MISC_MODIFIER" INT DEFAULT 0,
        "DESCRIPTION" VARCHAR(400),

        PRIMARY KEY (CHARACTER_ID, SKILL_ID),
        FOREIGN KEY (CHARACTER_ID) REFERENCES CHARACTER_RECORD(ID) ON DELETE CASCADE,
        FOREIGN KEY (SKILL_ID) REFERENCES SKILL_MASTER(ID)
   );

CREATE TABLE "CHARACTER_ABILITY_RECORD"
  (     "CHARACTER_ID" INT,
        "ABILITY_ID" INT,
        "BASE" INT DEFAULT 0,
        "FEAT_MODIFIER" INT DEFAULT 0,
        "MISC_MODIFIER" INT DEFAULT 0,
        "DESCRIPTION" VARCHAR(400),

        PRIMARY KEY (CHARACTER_ID, ABILITY_ID),
        FOREIGN KEY (CHARACTER_ID) REFERENCES CHARACTER_RECORD(ID) ON DELETE CASCADE,
        FOREIGN KEY (ABILITY_ID) REFERENCES ABILITY_MASTER(ID)
   );

CREATE TABLE "CHARACTER_SAVE_RECORD"
  (     "CHARACTER_ID" INT,
        "SAVE_ID" INT,
        "MISC_MODIFIER" INT,
        "DESCRIPTION" VARCHAR(400),

        PRIMARY KEY (CHARACTER_ID, SAVE_ID),
        FOREIGN KEY (CHARACTER_ID) REFERENCES CHARACTER_RECORD(ID) ON DELETE CASCADE,
        FOREIGN KEY (SAVE_ID) REFERENCES SAVE_MASTER(ID)
   );

-- // テンポラリ。実際には VARCHAR でなく、ITEM_ID のようなIDを見に行くようにする
CREATE TABLE "CHARACTER_EQUIPMENT"
  (     "CHARACTER_ID" INT,
        "HEAD"     VARCHAR(200),
        "CLOAK"    VARCHAR(200),
        "NECKLACE" VARCHAR(200),
        "ARM_RING" VARCHAR(200),
        "RING1"    VARCHAR(200),
        "RING2"  VARCHAR(200),
        "BELT"   VARCHAR(200),
        "ARMOR"  VARCHAR(200),
        "ARM1"   VARCHAR(200), -- 後々 int へ変換し、ARM_MASTER から選択する予定
        "ARM2"   VARCHAR(200), -- 後々 int へ変換し、ARM_MASTER から選択する予定
        "EYE"    VARCHAR(200),
        "ROBE"   VARCHAR(200),
        "VEST"   VARCHAR(200),
        "GLOVE"  VARCHAR(200),
        "SHIELD" VARCHAR(200),
        "BOOTS"  VARCHAR(200),
        "POCKET"   INT,
        "SKILL_ARMOR_MOD" INT,  -- 鎧ペナルティ値
        "SKILL_SHIELD_MOD" INT, -- 盾ペナルティ値
        "DEX_AC_ARMOR_LIMIT" INT,  -- 鎧敏捷力ボーナス上限
        "DEX_AC_SHIELD_LIMIT" INT, -- 盾敏捷力ボーナス上限

        PRIMARY KEY (CHARACTER_ID),
        FOREIGN KEY (CHARACTER_ID) REFERENCES CHARACTER_RECORD(ID) ON DELETE CASCADE
  );

CREATE TABLE "SKILL_SYNERGY_MASTER"
(       "SKILL_ID" INT,
        "AFFECTED_BY" INT,

        PRIMARY KEY ( SKILL_ID, AFFECTED_BY),
        FOREIGN KEY (SKILL_ID) REFERENCES SKILL_MASTER(ID)
);

CREATE TABLE "PLAYER_MASTER"
(     "ID" INT  GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
      "USERNAME" VARCHAR(100),
      "PASSWORD" VARCHAR(100),
      "PLAYER_NAME" VARCHAR(200)
);
